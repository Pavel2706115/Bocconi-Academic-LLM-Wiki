$$ \bar{t} = \frac{Y}{T} \ \text{average tax rate} \ \ \ t'=\frac{\delta T}{\delta Y} \ \text{marginal tax rate}
$$
$$ \text{How the average tax rate changes with the income: } \frac{\delta \bar{t}}{\delta Y} = \frac{\delta{(\frac{T}{Y})}}{\delta{Y}} = \frac{t'Y - T}{Y^2} = \frac{t' - \bar{t}}{Y}$$
$$ \text{Progressivity by tax credits} \ \ \ \ \  \begin{array}{l}T= tY - f \\ t' = t \\ \bar{t} = \frac{T}{Y} = \frac{tY-f}{Y} = t - \frac{f}{Y} \implies \bar{t} < t \end{array}
$$
$$ \text{Progressivity by deductions} \ \ \ \ \  \begin{array}{l}T= t(Y - d) \\ t' = t \\ \bar{t} = \frac{T}{Y} = \frac{t(Y-d)}{Y} = t - \frac{td}{Y} \implies \bar{t} < t \end{array}
$$
